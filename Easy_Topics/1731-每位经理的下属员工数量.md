3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One : 隐式自连接（多表查询）
    SELECT
        `E1`.`employee_id`,
        `E1`.`name`,
        COUNT(`E2`.`reports_to`) AS `reports_count`,
        ROUND(AVG(`E2`.`age`), 0) AS `average_age`
    FROM
        `Employees` `E1`, `Employees` `E2`
    WHERE
        `E1`.`employee_id` = `E2`.`reports_to`
    GROUP BY
        `E1`.`employee_id`
    ORDER BY
        `E1`.`employee_id`


    # ------------------------------
    # Method Two : 显式自连接（内连接）
    SELECT
        `E1`.`employee_id`,
        `E1`.`name`,
        COUNT(`E2`.`reports_to`) AS `reports_count`,
        ROUND(AVG(`E2`.`age`), 0) AS `average_age`
    FROM
        `Employees` AS `E1`
    INNER JOIN
        `Employees` AS `E2` ON `E1`.`employee_id` = `E2`.`reports_to`
    GROUP BY
        `E1`.`employee_id`
    ORDER BY
        `E1`.`employee_id`;
    ```
