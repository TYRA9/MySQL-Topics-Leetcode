3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One : DISTINCT
    SELECT
        DISTINCT `author_id` AS `id`
    FROM
        `Views`
    WHERE
        `author_id` = `viewer_id`
    ORDER BY
        `author_id`;

    # -------------------------
    # Method Two : GROUP BY
    SELECT
        `author_id` AS `id`
    FROM
        `Views`
    WHERE
        `author_id` = `viewer_id`
    GROUP BY
        `author_id`
    ORDER BY
        `author_id`;

    # -------------------------
    # Method Three : EXISTS
    # EXISTS is more suitable for multi-tables problems
    SELECT
        DISTINCT `V1`.`author_id` AS `id`
    FROM
        `Views` `V1`
    WHERE EXISTS (
        SELECT
            1
        FROM
            `Views` `V2`
        WHERE
            `V1`.`author_id` = `V2`.`author_id`
        AND
            `V1`.`author_id` = `V2`.`viewer_id`
    )
    ORDER BY
        `author_id`;
    ```
