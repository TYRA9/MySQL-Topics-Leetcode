    ```sql
    # Write your MySQL query statement below
    # Method One : Subquery
    -- While not strictly necessary for correctness, 
    -- including DISTINCT can often improve the performance of IN and NOT IN subqueries.
    SELECT
        `name`
    FROM
        `SalesPerson`
    WHERE
        `sales_id` NOT IN (
            SELECT 
                DISTINCT `o`.`sales_id`
            FROM
                `Orders` `o`
            INNER JOIN
                `Company` `c` ON `o`.`com_id` = `c`.`com_id`
            WHERE
                `c`.`name` = 'RED'
        );

    # -------------------------
    # Method Two : Subquery with LEFT JOIN
    SELECT
        `sp`.`name`
    FROM
        `SalesPerson` `sp`
    LEFT JOIN (
        SELECT
            DISTINCT `o`.`sales_id`
        FROM
            `Orders` `o`
        INNER JOIN 
            `Company` `c` ON `o`.`com_id` = `c`.`com_id`
        WHERE
            `c`.`name` = 'RED'
    ) AS `REDCompanyOrders` ON `sp`.`sales_id` = `REDCompanyOrders`.`sales_id`
    WHERE
        `REDCompanyOrders`.`sales_id` IS NULL;

    # -------------------------
    # Method Three : NOT EXISTS
    SELECT
        `sp`.`name`
    FROM 
        `SalesPerson` `sp`
    WHERE
        NOT EXISTS (
            SELECT
                1
            FROM
                `Orders` `o`
            INNER JOIN
                `Company` `c` ON `o`.`com_id` = `c`.`com_id`
            WHERE
                `c`.`name` = 'RED'
            AND 
                `sp`.`sales_id` = `o`.`sales_id`
        );
    ```
