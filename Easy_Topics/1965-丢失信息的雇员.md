3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: 外连接 + UNION ALL (合并查询)
    SELECT
        `E`.`employee_id`
    FROM
        `Employees` `E`
    LEFT JOIN
        `Salaries` `S` ON `E`.`employee_id` = `S`.`employee_id`
    WHERE
        `salary` IS NULL

    UNION ALL

    SELECT
        `S`.`employee_id`
    FROM
        `Employees` `E`
    RIGHT JOIN
        `Salaries` `S` ON `E`.`employee_id` = `S`.`employee_id`
    WHERE
        `name` IS NULL
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Two: NOT IN + UNION ALL
    SELECT
        `employee_id`
    FROM 
        `Employees`
    WHERE
        `employee_id` NOT IN ( SELECT `employee_id` FROM `Salaries` )

    UNION ALL

    SELECT
        `employee_id`
    FROM
        `Salaries`
    WHERE
        `employee_id` NOT IN ( SELECT `employee_id` FROM `Employees` )
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Three: Subquery (嵌套查询), Most Efficient
    -- (1) If an ID appears twice, it means it exists in both tables (information is complete).
    -- (2) If an ID appears exactly once, it means it is missing from one of the tables.
    SELECT
        `employee_id`
    FROM (
        SELECT `employee_id`
        FROM `Employees`

        UNION ALL

        SELECT `employee_id`
        FROM `Salaries`
    ) AS `Combined_IDs`
    GROUP BY
        `employee_id`
    HAVING
        COUNT(*) = 1
    ORDER BY
        `employee_id`;
    ```
