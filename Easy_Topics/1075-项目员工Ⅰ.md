    ```sql
    # Write your MySQL query statement below
    # Method One : GROUP BY + AVG
    SELECT
        `project_id`,
        ROUND(AVG(`experience_years`), 2) AS `average_years`
    FROM
        `Project`, `Employee`
    WHERE
        `Project`.`employee_id` = `Employee`.`employee_id`
    GROUP BY
        `project_id`;

    # -------------------------
    # Method Two : GROUP BY + (SUM + COUNT)
    SELECT
        `project_id`,
        ROUND(SUM(`experience_years`) / COUNT(`experience_years`), 2) AS `average_years`
    FROM
        `Project`, `Employee`
    WHERE
        `Project`.`employee_id` = `Employee`.`employee_id`
    GROUP BY
        `project_id`;

    # -------------------------
    # Method Three : INNER JOIN
    SELECT
        `project_id`,
        ROUND(AVG(`experience_years`), 2) AS `average_years`
    FROM
        `Project`
    INNER JOIN
        `Employee` USING(`employee_id`)
    GROUP BY
        `project_id`;

    # -------------------------
    # Method Four : Window Function
    -- Depiction :
    -- Unlike a GROUP BY clause, a window function does not collapse the rows. 
    -- Instead, it computes the aggregate value and returns it 
    -- alongside each row within its partition. So, if a project has 3 employees, 
    -- the calculated average for that project will appear on all 3 rows 
    -- corresponding to that project.
    SELECT 
        DISTINCT `P`.`project_id`,
        ROUND(AVG(`E`.`experience_years`) OVER (PARTITION BY project_id),2) AS average_years
    FROM 
        `Project` AS `P` 
    INNER JOIN 
        `Employee` AS `E` ON `P`.`employee_id` = `E`.`employee_id`;
    ```
