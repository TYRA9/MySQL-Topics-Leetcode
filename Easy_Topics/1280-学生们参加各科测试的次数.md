3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: CROSS JOIN + LEFT JOIN (GROUP BY Later)
    SELECT 
        `Stu`.`student_id`,
        `Stu`.`student_name`,
        `Sub`.`subject_name`,
        COUNT(`Exam`.`student_id`) AS `attended_exams`
    FROM
        `Students` `Stu`
    CROSS JOIN
        `Subjects` `Sub`
    LEFT JOIN
        `Examinations` `Exam` ON `Stu`.`student_id` = `Exam`.`student_id` AND `Sub`.`subject_name` = `Exam`.`subject_name`
    GROUP BY 
        `Stu`.`student_id`, `Stu`.`student_name`, `Sub`.`subject_name` -- All non-aggregated selected columns
    ORDER BY
        `Stu`.`student_id`, `Sub`.`subject_name`;

    # -------------------------
    # Method Two: GROUP BY Early
    SELECT
        `Stu`.`student_id`,
        `Stu`.`student_name`,
        `Sub`.`subject_name`,
        IFNULL(`GroupedExam`.`attended_exams`, 0) AS `attended_exams`
    FROM
        `Students` AS `Stu`
    CROSS JOIN
        `Subjects` AS `Sub`
    LEFT JOIN (
        SELECT 
            `student_id`,
            `subject_name`,
            COUNT(*) AS `attended_exams`
        FROM
            `Examinations`
        GROUP BY
            `student_id`, `subject_name`
    ) AS `GroupedExam` ON `Stu`.`student_id` = `GroupedExam`.`student_id` AND `Sub`.`subject_name` = `GroupedExam`.`subject_name`
    ORDER BY
        `Stu`.`student_id`, `Sub`.`subject_name`;
    ```
