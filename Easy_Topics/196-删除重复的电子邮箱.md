3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: 隐式自连接（多表查询）
    DELETE 
        `p1` 
    FROM
        `Person` `p1`, `Person` `p2`
    WHERE
        `p1`.`email` = `p2`.`email` 
    AND 
        `p1`.`id` > `p2`.`id`;

    # -------------------------
    # Method Two: 显式自连接（内连接）
    DELETE
        `p1`
    FROM
        `Person` `p1`
    INNER JOIN
        `Person` `p2` ON `p1`.`email` = `p2`.`email`
    WHERE
        `p1`.`id` > `p2`.`id`;

    # -------------------------
    # Method Three : 嵌套子查询
    DELETE FROM
        `Person`
    WHERE
        `id` NOT IN (
            SELECT 
                `id` 
            FROM (
                SELECT 
                    MIN(`id`) AS `id`
                FROM 
                    `Person`
                GROUP BY
                    `email`
            ) AS `temp_p`
        );
    ```
