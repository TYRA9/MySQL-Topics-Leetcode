3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One:  GROUP BY + aggregate function MIN()
    SELECT 
        `player_id`, 
        MIN(`event_date`) AS `first_login`
    FROM 
        `Activity`
    GROUP BY 
        `player_id`;

    # -------------------------
    # Method Two: Using Window Functions (MySQL 8.0+)
    SELECT
        `player_id`,
        `event_date` AS `first_login`
    FROM (
        SELECT 
            `player_id`,
            `event_date`,
            ROW_NUMBER() OVER (PARTITION BY `player_id` ORDER BY `event_date` ASC) AS `rn`
        FROM
            `Activity`
    ) AS `sub`
    WHERE
        `rn` = 1;

    # -------------------------
    # Method Three: Using LEFT JOIN / IS NULL (Anti-Join)
    SELECT
        `A1`.`player_id`,
        `A1`.`event_date` AS `first_login`
    FROM 
        `Activity` `A1`
    LEFT JOIN
        `Activity` `A2` ON `A1`.`player_id` = `A2`.`player_id` AND `A1`.`event_date` > `A2`.`event_date`
    WHERE
        `A2`.`event_date` IS NULL;
    ```
