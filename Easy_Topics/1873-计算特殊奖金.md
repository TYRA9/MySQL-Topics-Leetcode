3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: IF + STRCMP (Containing SUBSTRING)
    SELECT
        `employee_id`,
        IF(`employee_id` % 2 = 1 AND STRCMP(SUBSTRING(`name`, 1, 1), 'M') != 0, `salary`, 0) AS `bonus`
    FROM
        `Employees`
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Two: IF + STRCMP (Containing LEFT)
    SELECT
        `employee_id`,
        IF(`employee_id` % 2 = 1 AND STRCMP(LEFT(`name`, 1), 'M') != 0, `salary`, 0) AS `bonus`
    FROM
        `Employees`
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Three: IF + LEFT (directly)
    SELECT
        `employee_id`,
        IF(`employee_id` % 2 = 1 AND LEFT(`name`, 1) != 'M', `salary`, 0) AS `bonus`
    FROM
        `Employees`
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Four: CASE WHEN
    SELECT
        `employee_id`,
        CASE 
            WHEN (`employee_id` % 2 = 1 AND `name` NOT LIKE 'M%') THEN `salary`
            ELSE 0
        END AS `bonus`
    FROM
        `Employees`
    ORDER BY
        `employee_id`;

    # ------------------------------
    # Method Five: "Bitwise AND" + REGEXP
    -- The last bit of all odd numbers is always 1, 
    -- while for even numbers it’s 0. So, 
    -- when performing bitwise AND operation with 1, 
    -- odd numbers give 1, and even numbers give 0.

    SELECT
        `employee_id`,
        IF(`employee_id` & 1 = 1 AND `name` REGEXP '^[^M]', `salary`, 0) AS `bonus`
    FROM
        `Employees`
    ORDER BY
        `employee_id`;
    ```
