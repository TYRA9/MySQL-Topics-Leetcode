3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: OR
    SELECT 
        `name`,
        `population`,
        `area`
    FROM
        `World`
    WHERE
        `area` >= 3000000 OR `population` >= 25000000;

    # -------------------------
    # Method Two : UNION
    SELECT 
        `name`,
        `population`,
        `area`
    FROM
        `World`
    WHERE
        `area` >= 3000000

    UNION

    SELECT 
        `name`,
        `population`,
        `area`
    FROM
        `World`
    WHERE
        `population` >= 25000000;

    # -------------------------
    # Method Three: CASE WHEN
    SELECT 
        `name`,
        `population`,
        `area`
    FROM
        `World`
    WHERE (
        CASE 
            WHEN `population` >= 25000000 THEN 1
            WHEN  `area` >= 3000000 THEN 1
            ELSE 0
        END
    ) = 1;
    ```
