3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One : SubQuery
    SELECT
        MAX(`num`) AS `num`
    FROM (
        SELECT
            `num`
        FROM
            `MyNumbers`
        GROUP BY
            `num`
        HAVING 
            COUNT(`num`) = 1
    ) AS `tempNumbers`;

    # -------------------------
    # Method Two : NOT IN
    SELECT 
        MAX(`num`) AS `num`
    FROM
        `MyNumbers`
    WHERE
        `num` NOT IN (
            SELECT 
                `num`
            FROM
                `MyNumbers`
            GROUP BY
                `num`
            HAVING
                COUNT(*) > 1
        );

    # -------------------------
    # Method Three: ORDER BY + LIMIT
    SELECT (
        SELECT num
        FROM MyNumbers
        GROUP BY num
        HAVING COUNT(num) = 1
        ORDER BY num DESC
        LIMIT 1
    ) AS num;
    ```
