3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One: DATE_ADD or DATE_SUB
    SELECT
        `w2`.`id`
    FROM
        `Weather` `w1`, `Weather` `w2`
    WHERE
        `w2`.`recordDate` = DATE_ADD(`w1`.`recordDate`, INTERVAL 1 DAY)
    AND
        `w2`.`temperature` > `w1`.`temperature`;
        
    # -------------------------
    # Method Two: DATEDIFF
    SELECT  
        `w2`.`id`
    FROM
        `Weather` `w1`, `Weather` `w2`
    WHERE
        DATEDIFF(`w2`.`recordDate`, `w1`.`recordDate`) = 1
    AND
        `w2`.`temperature` > `w1`.`temperature`;


    # -------------------------
    # 显式自连接版本（通过内连接 INNER JOIN 来实现自连接）
    SELECT  
        `w2`.`id`
    FROM
        `Weather` `w1`
    INNER JOIN
        `Weather` `w2` ON `w2`.`recordDate` = DATE_ADD(`w1`.`recordDate`, INTERVAL 1 DAY)
    AND
        `w2`.`temperature` > `w1`.`temperature`;

    # -------------------------
    SELECT  
        `w2`.`id`
    FROM
        `Weather` `w1`
    INNER JOIN
        `Weather` `w2` ON DATEDIFF(`w2`.`recordDate`, `w1`.`recordDate`) = 1
    AND
        `w2`.`temperature` > `w1`.`temperature`;
    ```
