3. **答案**（纯代码）： 
      
    ```sql
    # Write your MySQL query statement below
    # Method One : INNER JOIN + GROUP BY
    SELECT
        `P`.`product_id`,
        `P`.`product_name`
    FROM
        `Product` `P`
    INNER JOIN
        `Sales` `S` USING(`product_id`)
    GROUP BY
        `P`.`product_id`, `P`.`product_name`
    HAVING
        MIN(`sale_date`) >= '2019-01-01' AND
        MAX(`sale_date`) <= '2019-03-31';

    # -------------------------
    # Method Two : SubQuery
    SELECT
        `product_id`, 
        `product_name`
    FROM
        `Product`
    WHERE
        `Product`.`product_id` IN (
            SELECT
                DISTINCT `product_id`
            FROM
                `Sales`
        )
    AND
        `Product`.`product_id` NOT IN (
            SELECT
                `P_sub`.`product_id`
            FROM
                `Product` `P_sub`
            INNER JOIN
                `Sales` `S_sub` USING (`product_id`)
            WHERE
                `S_sub`.`sale_date` NOT BETWEEN '2019-01-01' AND '2019-03-31'
        );
    ```
